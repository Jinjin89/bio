git examples

  1. git globa options

    # show global options
git config --list

# set git editor
git config --global core.editor "code --wait"
git config -e 
git config --global -e

  2. get git status short

    git status 
git status -s

  3. create new branch

    # create new brance
git checkout -b <newbranch>

# back to mater
git checkout master

# merge new brache into master(should be in master branche)
git merge <newbranch>

  4. clone with ssh

    git clone username@ip_address:/path/to/your/git

  5. build the data

    # 1) build new data
git init

# 2) rename branch
git rename -m master current

# 3) add remote data
git remote add <repo-name> username@ip_address:/path/to/your/git

# 4) pull or push
git pull 
git push
